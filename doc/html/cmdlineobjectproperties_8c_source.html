253 check_possible_values(const char *val, const char *values[])
254 {
255  int i, found, last;
256  size_t len;
257 
258  if (!val) /* otherwise strlen() crashes below */
259  return -1; /* -1 means no argument for the option */
260 
261  found = last = 0;
262 
263  for (i = 0, len = strlen(val); values[i]; ++i)

547 struct custom_getopt_data {
548  /*
549  * These have exactly the same meaning as the corresponding global variables,
550  * except that they are used for the reentrant versions of getopt.
551  */
552  int custom_optind;
553  int custom_opterr;
554  int custom_optopt;
555  char *custom_optarg;
556 
557  /* True if the internal members have been initialized. */
558  int initialized;
559 
560  /*
561  * The next char to be scanned in the option-element in which the last option
562  * character we returned was found. This allows us to pick up the scan where
563  * we left off. If this is zero, or a null string, it means resume the scan by
564  * advancing to the next ARGV-element.
565  */
566  char *nextchar;
567 
568  /*
569  * Describe the part of ARGV that contains non-options that have been skipped.
570  * `first_nonopt' is the index in ARGV of the first of them; `last_nonopt' is
571  * the index after the last of them.
572  */
573  int first_nonopt;
574  int last_nonopt;

577 /*
578  * the variables optarg, optind, opterr and optopt are renamed with
579  * the custom_ prefix so that they don't interfere with getopt ones.
580  *
581  * Moreover they're static so they are visible only from within the
582  * file where this very file will be included.
583  */
584 
585 /*
586  * For communication from `custom_getopt' to the caller. When `custom_getopt' finds an
587  * option that takes an argument, the argument value is returned here.
588  */
589 static char *custom_optarg;
590 
591 /*
592  * Index in ARGV of the next element to be scanned. This is used for
593  * communication to and from the caller and for communication between
594  * successive calls to `custom_getopt'.
595  *
596  * On entry to `custom_getopt', 1 means this is the first call; initialize.
597  *
598  * When `custom_getopt' returns -1, this is the index of the first of the non-option
599  * elements that the caller should itself scan.
600  *
601  * Otherwise, `custom_optind' communicates from one call to the next how much of ARGV
602  * has been scanned so far.
603  *
604  * 1003.2 says this must be 1 before any call.
605  */
606 static int custom_optind = 1;
607 
608 /*
609  * Callers store zero here to inhibit the error message for unrecognized
610  * options.
611  */
612 static int custom_opterr = 1;
613 
614 /*
615  * Set to an option character which was unrecognized. This must be initialized
616  * on some systems to avoid linking in the system's own getopt implementation.
617  */
618 static int custom_optopt = '?';
619 
620 /*
621  * Exchange two adjacent subsequences of ARGV. One subsequence is elements
622  * [first_nonopt,last_nonopt) which contains all the non-options that have been
623  * skipped so far. The other is elements [last_nonopt,custom_optind), which contains
624  * all the options processed since those non-options were skipped.
625  * `first_nonopt' and `last_nonopt' are relocated so that they describe the new
626  * indices of the non-options in ARGV after they are moved.
627  */
628 static void exchange(char **argv, struct custom_getopt_data *d)
629 {
630  int bottom = d->first_nonopt;
631  int middle = d->last_nonopt;
632  int top = d->custom_optind;
633  char *tem;
634 
635  /*
636  * Exchange the shorter segment with the far end of the longer segment.
637  * That puts the shorter segment into the right place. It leaves the
638  * longer segment in the right place overall, but it consists of two
639  * parts that need to be swapped next.
640  */
641  while (top > middle && middle > bottom) {
642  if (top - middle > middle - bottom) {
643  /* Bottom segment is the short one. */
644  int len = middle - bottom;
645  int i;
646 
647  /* Swap it with the top part of the top segment. */
648  for (i = 0; i < len; i++) {
649  tem = argv[bottom + i];
650  argv[bottom + i] =
651  argv[top - (middle - bottom) + i];
652  argv[top - (middle - bottom) + i] = tem;
653  }
654  /* Exclude the moved bottom segment from further swapping. */
655  top -= len;
656  } else {
657  /* Top segment is the short one. */
658  int len = top - middle;
659  int i;
660 
661  /* Swap it with the bottom part of the bottom segment. */
662  for (i = 0; i < len; i++) {
663  tem = argv[bottom + i];
664  argv[bottom + i] = argv[middle + i];
665  argv[middle + i] = tem;
666  }
667  /* Exclude the moved top segment from further swapping. */
668  bottom += len;
669  }
670  }
671  /* Update records for the slots the non-options now occupy. */
672  d->first_nonopt += (d->custom_optind - d->last_nonopt);
673  d->last_nonopt = d->custom_optind;
674 }

677 static void custom_getopt_initialize(struct custom_getopt_data *d)
678 {
679  /*
680  * Start processing options with ARGV-element 1 (since ARGV-element 0
681  * is the program name); the sequence of previously skipped non-option
682  * ARGV-elements is empty.
683  */
684  d->first_nonopt = d->last_nonopt = d->custom_optind;
685  d->nextchar = NULL;
686  d->initialized = 1;
687 }
688 
689 #define NONOPTION_P (argv[d->custom_optind][0] != '-' || argv[d->custom_optind][1] == '\0')
690 
691 /* return: zero: continue, nonzero: return given value to user */
692 static int shuffle_argv(int argc, char *const *argv,const struct option *longopts,
693  struct custom_getopt_data *d)
694 {
695  /*
696  * Give FIRST_NONOPT & LAST_NONOPT rational values if CUSTOM_OPTIND has been
697  * moved back by the user (who may also have changed the arguments).
698  */
699  if (d->last_nonopt > d->custom_optind)
700  d->last_nonopt = d->custom_optind;
701  if (d->first_nonopt > d->custom_optind)
702  d->first_nonopt = d->custom_optind;
703  /*
704  * If we have just processed some options following some
705  * non-options, exchange them so that the options come first.
706  */
707  if (d->first_nonopt != d->last_nonopt &&
708  d->last_nonopt != d->custom_optind)
709  exchange((char **) argv, d);
710  else if (d->last_nonopt != d->custom_optind)
711  d->first_nonopt = d->custom_optind;
712  /*
713  * Skip any additional non-options and extend the range of
714  * non-options previously skipped.
715  */
716  while (d->custom_optind < argc && NONOPTION_P)
717  d->custom_optind++;
718  d->last_nonopt = d->custom_optind;
719  /*
720  * The special ARGV-element `--' means premature end of options. Skip
721  * it like a null option, then exchange with previous non-options as if
722  * it were an option, then skip everything else like a non-option.
723  */
724  if (d->custom_optind != argc && !strcmp(argv[d->custom_optind], "--")) {
725  d->custom_optind++;
726  if (d->first_nonopt != d->last_nonopt
727  && d->last_nonopt != d->custom_optind)
728  exchange((char **) argv, d);
729  else if (d->first_nonopt == d->last_nonopt)
730  d->first_nonopt = d->custom_optind;
731  d->last_nonopt = argc;
732  d->custom_optind = argc;
733  }
734  /*
735  * If we have done all the ARGV-elements, stop the scan and back over
736  * any non-options that we skipped and permuted.
737  */
738  if (d->custom_optind == argc) {
739  /*
740  * Set the next-arg-index to point at the non-options that we
741  * previously skipped, so the caller will digest them.
742  */
743  if (d->first_nonopt != d->last_nonopt)
744  d->custom_optind = d->first_nonopt;
745  return -1;
746  }
747  /*
748  * If we have come to a non-option and did not permute it, either stop
749  * the scan or describe it to the caller and pass it by.
750  */
751  if (NONOPTION_P) {
752  d->custom_optarg = argv[d->custom_optind++];
753  return 1;
754  }
755  /*
756  * We have found another option-ARGV-element. Skip the initial
757  * punctuation.
758  */
759  d->nextchar = (argv[d->custom_optind] + 1 + (longopts != NULL && argv[d->custom_optind][1] == '-'));
760  return 0;
761 }
762 
763 /*
764  * Check whether the ARGV-element is a long option.
765  *
766  * If there's a long option "fubar" and the ARGV-element is "-fu", consider
767  * that an abbreviation of the long option, just like "--fu", and not "-f" with
768  * arg "u".
769  *
770  * This distinction seems to be the most useful approach.
771  *
772  */
773 static int check_long_opt(int argc, char *const *argv, const char *optstring,
774  const struct option *longopts, int *longind,
775  int print_errors, struct custom_getopt_data *d)
776 {
777  char *nameend;
778  const struct option *p;
779  const struct option *pfound = NULL;
780  int exact = 0;
781  int ambig = 0;
782  int indfound = -1;
783  int option_index;
784 
785  for (nameend = d->nextchar; *nameend && *nameend != '='; nameend++)
786  /* Do nothing. */ ;
787 
788  /* Test all long options for either exact match or abbreviated matches */
789  for (p = longopts, option_index = 0; p->name; p++, option_index++)
790  if (!strncmp(p->name, d->nextchar, nameend - d->nextchar)) {
791  if ((unsigned int) (nameend - d->nextchar)
792  == (unsigned int) strlen(p->name)) {
793  /* Exact match found. */
794  pfound = p;
795  indfound = option_index;
796  exact = 1;
797  break;
798  } else if (pfound == NULL) {
799  /* First nonexact match found. */
800  pfound = p;
801  indfound = option_index;
802  } else if (pfound->has_arg != p->has_arg
803  || pfound->flag != p->flag
804  || pfound->val != p->val)
805  /* Second or later nonexact match found. */
806  ambig = 1;
807  }
808  if (ambig && !exact) {
809  if (print_errors) {
810  fprintf(stderr,
811  "%s: option `%s' is ambiguous\n",
812  argv[0], argv[d->custom_optind]);
813  }
814  d->nextchar += strlen(d->nextchar);
815  d->custom_optind++;
816  d->custom_optopt = 0;
817  return '?';
818  }
819  if (pfound) {
820  option_index = indfound;
821  d->custom_optind++;
822  if (*nameend) {
823  if (pfound->has_arg != no_argument)
824  d->custom_optarg = nameend + 1;
825  else {
826  if (print_errors) {
827  if (argv[d->custom_optind - 1][1] == '-') {
828  /* --option */
829  fprintf(stderr, "%s: option `--%s' doesn't allow an argument\n",
830  argv[0], pfound->name);
831  } else {
832  /* +option or -option */
833  fprintf(stderr, "%s: option `%c%s' doesn't allow an argument\n",
834  argv[0], argv[d->custom_optind - 1][0], pfound->name);
835  }
836 
837  }
838  d->nextchar += strlen(d->nextchar);
839  d->custom_optopt = pfound->val;
840  return '?';
841  }
842  } else if (pfound->has_arg == required_argument) {
843  if (d->custom_optind < argc)
844  d->custom_optarg = argv[d->custom_optind++];
845  else {
846  if (print_errors) {
847  fprintf(stderr,
848  "%s: option `%s' requires an argument\n",
849  argv[0],
850  argv[d->custom_optind - 1]);
851  }
852  d->nextchar += strlen(d->nextchar);
853  d->custom_optopt = pfound->val;
854  return optstring[0] == ':' ? ':' : '?';
855  }
856  }
857  d->nextchar += strlen(d->nextchar);
858  if (longind != NULL)
859  *longind = option_index;
860  if (pfound->flag) {
861  *(pfound->flag) = pfound->val;
862  return 0;
863  }
864  return pfound->val;
865  }
866  /*
867  * Can't find it as a long option. If this is not getopt_long_only, or
868  * the option starts with '--' or is not a valid short option, then
869  * it's an error. Otherwise interpret it as a short option.
870  */
871  if (print_errors) {
872  if (argv[d->custom_optind][1] == '-') {
873  /* --option */
874  fprintf(stderr,
875  "%s: unrecognized option `--%s'\n",
876  argv[0], d->nextchar);
877  } else {
878  /* +option or -option */
879  fprintf(stderr,
880  "%s: unrecognized option `%c%s'\n",
881  argv[0], argv[d->custom_optind][0],
882  d->nextchar);
883  }
884  }
885  d->nextchar = (char *) "";
886  d->custom_optind++;
887  d->custom_optopt = 0;
888  return '?';
889 }

891 static int check_short_opt(int argc, char *const *argv, const char *optstring,
892  int print_errors, struct custom_getopt_data *d)
893 {
894  char c = *d->nextchar++;
895  const char *temp = strchr(optstring, c);
896 
897  /* Increment `custom_optind' when we start to process its last character. */
898  if (*d->nextchar == '\0')
899  ++d->custom_optind;
900  if (!temp || c == ':') {
901  if (print_errors)
902  fprintf(stderr, "%s: invalid option -- %c\n", argv[0], c);
903 
904  d->custom_optopt = c;
905  return '?';
906  }
907  if (temp[1] == ':') {
908  if (temp[2] == ':') {
909  /* This is an option that accepts an argument optionally. */
910  if (*d->nextchar != '\0') {
911  d->custom_optarg = d->nextchar;
912  d->custom_optind++;
913  } else
914  d->custom_optarg = NULL;
915  d->nextchar = NULL;
916  } else {
917  /* This is an option that requires an argument. */
918  if (*d->nextchar != '\0') {
919  d->custom_optarg = d->nextchar;
920  /*
921  * If we end this ARGV-element by taking the
922  * rest as an arg, we must advance to the next
923  * element now.
924  */
925  d->custom_optind++;
926  } else if (d->custom_optind == argc) {
927  if (print_errors) {
928  fprintf(stderr,
929  "%s: option requires an argument -- %c\n",
930  argv[0], c);
931  }
932  d->custom_optopt = c;
933  if (optstring[0] == ':')
934  c = ':';
935  else
936  c = '?';
937  } else
938  /*
939  * We already incremented `custom_optind' once;
940  * increment it again when taking next ARGV-elt
941  * as argument.
942  */
943  d->custom_optarg = argv[d->custom_optind++];
944  d->nextchar = NULL;
945  }
946  }
947  return c;
948 }

950 /*
951  * Scan elements of ARGV for option characters given in OPTSTRING.
952  *
953  * If an element of ARGV starts with '-', and is not exactly "-" or "--",
954  * then it is an option element. The characters of this element
955  * (aside from the initial '-') are option characters. If `getopt'
956  * is called repeatedly, it returns successively each of the option characters
957  * from each of the option elements.
958  *
959  * If `getopt' finds another option character, it returns that character,
960  * updating `custom_optind' and `nextchar' so that the next call to `getopt' can
961  * resume the scan with the following option character or ARGV-element.
962  *
963  * If there are no more option characters, `getopt' returns -1.
964  * Then `custom_optind' is the index in ARGV of the first ARGV-element
965  * that is not an option. (The ARGV-elements have been permuted
966  * so that those that are not options now come last.)
967  *
968  * OPTSTRING is a string containing the legitimate option characters.
969  * If an option character is seen that is not listed in OPTSTRING,
970  * return '?' after printing an error message. If you set `custom_opterr' to
971  * zero, the error message is suppressed but we still return '?'.
972  *
973  * If a char in OPTSTRING is followed by a colon, that means it wants an arg,
974  * so the following text in the same ARGV-element, or the text of the following
975  * ARGV-element, is returned in `custom_optarg'. Two colons mean an option that
976  * wants an optional arg; if there is text in the current ARGV-element,
977  * it is returned in `custom_optarg', otherwise `custom_optarg' is set to zero.
978  *
979  * If OPTSTRING starts with `-' or `+', it requests different methods of
980  * handling the non-option ARGV-elements.
981  * See the comments about RETURN_IN_ORDER and REQUIRE_ORDER, above.
982  *
983  * Long-named options begin with `--' instead of `-'.
984  * Their names may be abbreviated as long as the abbreviation is unique
985  * or is an exact match for some defined option. If they have an
986  * argument, it follows the option name in the same ARGV-element, separated
987  * from the option name by a `=', or else the in next ARGV-element.
988  * When `getopt' finds a long-named option, it returns 0 if that option's
989  * `flag' field is nonzero, the value of the option's `val' field
990  * if the `flag' field is zero.
991  *
992  * The elements of ARGV aren't really const, because we permute them.
993  * But we pretend they're const in the prototype to be compatible
994  * with other systems.
995  *
996  * LONGOPTS is a vector of `struct option' terminated by an
997  * element containing a name which is zero.
998  *
999  * LONGIND returns the index in LONGOPT of the long-named option found.
1000  * It is only valid when a long-named option has been found by the most
1001  * recent call.
1002  *
1003  * Return the option character from OPTS just read. Return -1 when there are
1004  * no more options. For unrecognized options, or options missing arguments,
1005  * `custom_optopt' is set to the option letter, and '?' is returned.
1006  *
1007  * The OPTS string is a list of characters which are recognized option letters,
1008  * optionally followed by colons, specifying that that letter takes an
1009  * argument, to be placed in `custom_optarg'.
1010  *
1011  * If a letter in OPTS is followed by two colons, its argument is optional.
1012  * This behavior is specific to the GNU `getopt'.
1013  *
1014  * The argument `--' causes premature termination of argument scanning,
1015  * explicitly telling `getopt' that there are no more options. If OPTS begins
1016  * with `--', then non-option arguments are treated as arguments to the option
1017  * '\0'. This behavior is specific to the GNU `getopt'.
1018  */
1019 
1020 static int getopt_internal_r(int argc, char *const *argv, const char *optstring,
1021  const struct option *longopts, int *longind,
1022  struct custom_getopt_data *d)
1023 {
1024  int ret, print_errors = d->custom_opterr;
1025 
1026  if (optstring[0] == ':')
1027  print_errors = 0;
1028  if (argc < 1)
1029  return -1;
1030  d->custom_optarg = NULL;
1031 
1032  /*
1033  * This is a big difference with GNU getopt, since optind == 0
1034  * means initialization while here 1 means first call.
1035  */
1036  if (d->custom_optind == 0 || !d->initialized) {
1037  if (d->custom_optind == 0)
1038  d->custom_optind = 1; /* Don't scan ARGV[0], the program name. */
1039  custom_getopt_initialize(d);
1040  }
1041  if (d->nextchar == NULL || *d->nextchar == '\0') {
1042  ret = shuffle_argv(argc, argv, longopts, d);
1043  if (ret)
1044  return ret;
1045  }
1046  if (longopts && (argv[d->custom_optind][1] == '-' ))
1047  return check_long_opt(argc, argv, optstring, longopts,
1048  longind, print_errors, d);
1049  return check_short_opt(argc, argv, optstring, print_errors, d);
1050 }
