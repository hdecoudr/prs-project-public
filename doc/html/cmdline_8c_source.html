617  As `getopt' works, it permutes the elements of ARGV so that,

662  * we left off. If this is zero, or a null string, it means resume the scan by

677  * the variables optarg, optind, opterr and optopt are renamed with
678  * the custom_ prefix so that they don't interfere with getopt ones.

684 /*
685  * For communication from `custom_getopt' to the caller. When `custom_getopt' finds an
686  * option that takes an argument, the argument value is returned here.
687  */
688 static char *custom_optarg;

692  * communication to and from the caller and for communication between
693  * successive calls to `custom_getopt'.
694  *
695  * On entry to `custom_getopt', 1 means this is the first call; initialize.
696  *
697  * When `custom_getopt' returns -1, this is the index of the first of the non-option

701  * has been scanned so far.
702  *

717 static int custom_optopt = '?';

719 /*
720  * Exchange two adjacent subsequences of ARGV. One subsequence is elements
721  * [first_nonopt,last_nonopt) which contains all the non-options that have been
722  * skipped so far. The other is elements [last_nonopt,custom_optind), which contains
723  * all the options processed since those non-options were skipped.
724  * `first_nonopt' and `last_nonopt' are relocated so that they describe the new
725  * indices of the non-options in ARGV after they are moved.
726  */
727 static void exchange(char **argv, struct custom_getopt_data *d)
728 {
729  int bottom = d->first_nonopt;
730  int middle = d->last_nonopt;
731  int top = d->custom_optind;
732  char *tem;
733 
734  /*
735  * Exchange the shorter segment with the far end of the longer segment.
736  * That puts the shorter segment into the right place. It leaves the
737  * longer segment in the right place overall, but it consists of two
738  * parts that need to be swapped next.
739  */
740  while (top > middle && middle > bottom) {
741  if (top - middle > middle - bottom) {
742  /* Bottom segment is the short one. */
743  int len = middle - bottom;
744  int i;
745 
746  /* Swap it with the top part of the top segment. */
747  for (i = 0; i < len; i++) {
748  tem = argv[bottom + i];
749  argv[bottom + i] =
750  argv[top - (middle - bottom) + i];
751  argv[top - (middle - bottom) + i] = tem;
752  }
753  /* Exclude the moved bottom segment from further swapping. */
754  top -= len;
755  } else {
756  /* Top segment is the short one. */
757  int len = top - middle;
758  int i;
759 
760  /* Swap it with the bottom part of the bottom segment. */
761  for (i = 0; i < len; i++) {
762  tem = argv[bottom + i];
763  argv[bottom + i] = argv[middle + i];
764  argv[middle + i] = tem;
765  }
766  /* Exclude the moved top segment from further swapping. */
767  bottom += len;
768  }
769  }
770  /* Update records for the slots the non-options now occupy. */
771  d->first_nonopt += (d->custom_optind - d->last_nonopt);
772  d->last_nonopt = d->custom_optind;
773 }

775 /* Initialize the internal data when the first call is made. */
776 static void custom_getopt_initialize(struct custom_getopt_data *d)
777 {
778  /*
779  * Start processing options with ARGV-element 1 (since ARGV-element 0
780  * is the program name); the sequence of previously skipped non-option
781  * ARGV-elements is empty.
782  */
783  d->first_nonopt = d->last_nonopt = d->custom_optind;
784  d->nextchar = NULL;
785  d->initialized = 1;
786 }
787 
788 #define NONOPTION_P (argv[d->custom_optind][0] != '-' || argv[d->custom_optind][1] == '\0')
789 
790 /* return: zero: continue, nonzero: return given value to user */
791 static int shuffle_argv(int argc, char *const *argv,const struct option *longopts,
792  struct custom_getopt_data *d)
793 {
794  /*
795  * Give FIRST_NONOPT & LAST_NONOPT rational values if CUSTOM_OPTIND has been
796  * moved back by the user (who may also have changed the arguments).
797  */
798  if (d->last_nonopt > d->custom_optind)
799  d->last_nonopt = d->custom_optind;
800  if (d->first_nonopt > d->custom_optind)
801  d->first_nonopt = d->custom_optind;
802  /*
803  * If we have just processed some options following some
804  * non-options, exchange them so that the options come first.
805  */
806  if (d->first_nonopt != d->last_nonopt &&
807  d->last_nonopt != d->custom_optind)
808  exchange((char **) argv, d);
809  else if (d->last_nonopt != d->custom_optind)
810  d->first_nonopt = d->custom_optind;
811  /*
812  * Skip any additional non-options and extend the range of
813  * non-options previously skipped.
814  */
815  while (d->custom_optind < argc && NONOPTION_P)
816  d->custom_optind++;
817  d->last_nonopt = d->custom_optind;
818  /*
819  * The special ARGV-element `--' means premature end of options. Skip
820  * it like a null option, then exchange with previous non-options as if
821  * it were an option, then skip everything else like a non-option.
822  */
823  if (d->custom_optind != argc && !strcmp(argv[d->custom_optind], "--")) {
824  d->custom_optind++;
825  if (d->first_nonopt != d->last_nonopt
826  && d->last_nonopt != d->custom_optind)
827  exchange((char **) argv, d);
828  else if (d->first_nonopt == d->last_nonopt)
829  d->first_nonopt = d->custom_optind;
830  d->last_nonopt = argc;
831  d->custom_optind = argc;
832  }
833  /*
834  * If we have done all the ARGV-elements, stop the scan and back over
835  * any non-options that we skipped and permuted.
836  */
837  if (d->custom_optind == argc) {
838  /*
839  * Set the next-arg-index to point at the non-options that we
840  * previously skipped, so the caller will digest them.
841  */
842  if (d->first_nonopt != d->last_nonopt)
843  d->custom_optind = d->first_nonopt;
844  return -1;
845  }
846  /*
847  * If we have come to a non-option and did not permute it, either stop
848  * the scan or describe it to the caller and pass it by.
849  */
850  if (NONOPTION_P) {
851  d->custom_optarg = argv[d->custom_optind++];
852  return 1;
853  }
854  /*
855  * We have found another option-ARGV-element. Skip the initial
856  * punctuation.
857  */
858  d->nextchar = (argv[d->custom_optind] + 1 + (longopts != NULL && argv[d->custom_optind][1] == '-'));
859  return 0;
860 }

862 /*
863  * Check whether the ARGV-element is a long option.
864  *
865  * If there's a long option "fubar" and the ARGV-element is "-fu", consider
866  * that an abbreviation of the long option, just like "--fu", and not "-f" with
867  * arg "u".
868  *
869  * This distinction seems to be the most useful approach.
870  *
871  */
872 static int check_long_opt(int argc, char *const *argv, const char *optstring,
873  const struct option *longopts, int *longind,
874  int print_errors, struct custom_getopt_data *d)
875 {
876  char *nameend;
877  const struct option *p;
878  const struct option *pfound = NULL;
879  int exact = 0;
880  int ambig = 0;
881  int indfound = -1;
882  int option_index;
883 
884  for (nameend = d->nextchar; *nameend && *nameend != '='; nameend++)
885  /* Do nothing. */ ;
886 
887  /* Test all long options for either exact match or abbreviated matches */
888  for (p = longopts, option_index = 0; p->name; p++, option_index++)
889  if (!strncmp(p->name, d->nextchar, nameend - d->nextchar)) {
890  if ((unsigned int) (nameend - d->nextchar)
891  == (unsigned int) strlen(p->name)) {
892  /* Exact match found. */
893  pfound = p;
894  indfound = option_index;
895  exact = 1;
896  break;
897  } else if (pfound == NULL) {
898  /* First nonexact match found. */
899  pfound = p;
900  indfound = option_index;
901  } else if (pfound->has_arg != p->has_arg
902  || pfound->flag != p->flag
903  || pfound->val != p->val)
904  /* Second or later nonexact match found. */
905  ambig = 1;
906  }
907  if (ambig && !exact) {
908  if (print_errors) {
909  fprintf(stderr,
910  "%s: option `%s' is ambiguous\n",
911  argv[0], argv[d->custom_optind]);
912  }
913  d->nextchar += strlen(d->nextchar);
914  d->custom_optind++;
915  d->custom_optopt = 0;
916  return '?';
917  }
918  if (pfound) {
919  option_index = indfound;
920  d->custom_optind++;
921  if (*nameend) {
922  if (pfound->has_arg != no_argument)
923  d->custom_optarg = nameend + 1;
924  else {
925  if (print_errors) {
926  if (argv[d->custom_optind - 1][1] == '-') {
927  /* --option */
928  fprintf(stderr, "%s: option `--%s' doesn't allow an argument\n",
929  argv[0], pfound->name);
930  } else {
931  /* +option or -option */
932  fprintf(stderr, "%s: option `%c%s' doesn't allow an argument\n",
933  argv[0], argv[d->custom_optind - 1][0], pfound->name);
934  }
935 
936  }
937  d->nextchar += strlen(d->nextchar);
938  d->custom_optopt = pfound->val;
939  return '?';
940  }
941  } else if (pfound->has_arg == required_argument) {
942  if (d->custom_optind < argc)
943  d->custom_optarg = argv[d->custom_optind++];
944  else {
945  if (print_errors) {
946  fprintf(stderr,
947  "%s: option `%s' requires an argument\n",
948  argv[0],
949  argv[d->custom_optind - 1]);
950  }
951  d->nextchar += strlen(d->nextchar);
952  d->custom_optopt = pfound->val;
953  return optstring[0] == ':' ? ':' : '?';
954  }
955  }
956  d->nextchar += strlen(d->nextchar);
957  if (longind != NULL)
958  *longind = option_index;
959  if (pfound->flag) {
960  *(pfound->flag) = pfound->val;
961  return 0;
962  }
963  return pfound->val;
964  }
965  /*
966  * Can't find it as a long option. If this is not getopt_long_only, or
967  * the option starts with '--' or is not a valid short option, then
968  * it's an error. Otherwise interpret it as a short option.
969  */
970  if (print_errors) {
971  if (argv[d->custom_optind][1] == '-') {
972  /* --option */
973  fprintf(stderr,
974  "%s: unrecognized option `--%s'\n",
975  argv[0], d->nextchar);
976  } else {
977  /* +option or -option */
978  fprintf(stderr,
979  "%s: unrecognized option `%c%s'\n",
980  argv[0], argv[d->custom_optind][0],
981  d->nextchar);
982  }
983  }
984  d->nextchar = (char *) "";
985  d->custom_optind++;
986  d->custom_optopt = 0;
987  return '?';
988 }

990 static int check_short_opt(int argc, char *const *argv, const char *optstring,
991  int print_errors, struct custom_getopt_data *d)
992 {
993  char c = *d->nextchar++;
994  const char *temp = strchr(optstring, c);
995 
996  /* Increment `custom_optind' when we start to process its last character. */
997  if (*d->nextchar == '\0')
998  ++d->custom_optind;
999  if (!temp || c == ':') {
1000  if (print_errors)
1001  fprintf(stderr, "%s: invalid option -- %c\n", argv[0], c);
1002 
1003  d->custom_optopt = c;
1004  return '?';
1005  }
1006  if (temp[1] == ':') {
1007  if (temp[2] == ':') {
1008  /* This is an option that accepts an argument optionally. */
1009  if (*d->nextchar != '\0') {
1010  d->custom_optarg = d->nextchar;
1011  d->custom_optind++;
1012  } else
1013  d->custom_optarg = NULL;
1014  d->nextchar = NULL;
1015  } else {
1016  /* This is an option that requires an argument. */
1017  if (*d->nextchar != '\0') {
1018  d->custom_optarg = d->nextchar;
1019  /*
1020  * If we end this ARGV-element by taking the
1021  * rest as an arg, we must advance to the next
1022  * element now.
1023  */
1024  d->custom_optind++;
1025  } else if (d->custom_optind == argc) {
1026  if (print_errors) {
1027  fprintf(stderr,
1028  "%s: option requires an argument -- %c\n",
1029  argv[0], c);
1030  }
1031  d->custom_optopt = c;
1032  if (optstring[0] == ':')
1033  c = ':';
1034  else
1035  c = '?';
1036  } else
1037  /*
1038  * We already incremented `custom_optind' once;
1039  * increment it again when taking next ARGV-elt
1040  * as argument.
1041  */
1042  d->custom_optarg = argv[d->custom_optind++];
1043  d->nextchar = NULL;
1044  }
1045  }
1046  return c;
1047 }

1049 /*
1050  * Scan elements of ARGV for option characters given in OPTSTRING.
1051  *
1052  * If an element of ARGV starts with '-', and is not exactly "-" or "--",
1053  * then it is an option element. The characters of this element
1054  * (aside from the initial '-') are option characters. If `getopt'
1055  * is called repeatedly, it returns successively each of the option characters
1056  * from each of the option elements.
1057  *
1058  * If `getopt' finds another option character, it returns that character,
1059  * updating `custom_optind' and `nextchar' so that the next call to `getopt' can
1060  * resume the scan with the following option character or ARGV-element.
1061  *
1062  * If there are no more option characters, `getopt' returns -1.
1063  * Then `custom_optind' is the index in ARGV of the first ARGV-element
1064  * that is not an option. (The ARGV-elements have been permuted
1065  * so that those that are not options now come last.)
1066  *
1067  * OPTSTRING is a string containing the legitimate option characters.
1068  * If an option character is seen that is not listed in OPTSTRING,
1069  * return '?' after printing an error message. If you set `custom_opterr' to
1070  * zero, the error message is suppressed but we still return '?'.
1071  *
1072  * If a char in OPTSTRING is followed by a colon, that means it wants an arg,
1073  * so the following text in the same ARGV-element, or the text of the following
1074  * ARGV-element, is returned in `custom_optarg'. Two colons mean an option that
1075  * wants an optional arg; if there is text in the current ARGV-element,
1076  * it is returned in `custom_optarg', otherwise `custom_optarg' is set to zero.
1077  *
1078  * If OPTSTRING starts with `-' or `+', it requests different methods of
1079  * handling the non-option ARGV-elements.
1080  * See the comments about RETURN_IN_ORDER and REQUIRE_ORDER, above.
1081  *
1082  * Long-named options begin with `--' instead of `-'.
1083  * Their names may be abbreviated as long as the abbreviation is unique
1084  * or is an exact match for some defined option. If they have an
1085  * argument, it follows the option name in the same ARGV-element, separated
1086  * from the option name by a `=', or else the in next ARGV-element.
1087  * When `getopt' finds a long-named option, it returns 0 if that option's
1088  * `flag' field is nonzero, the value of the option's `val' field
1089  * if the `flag' field is zero.
1090  *
1091  * The elements of ARGV aren't really const, because we permute them.
1092  * But we pretend they're const in the prototype to be compatible
1093  * with other systems.
1094  *
1095  * LONGOPTS is a vector of `struct option' terminated by an
1096  * element containing a name which is zero.
1097  *
1098  * LONGIND returns the index in LONGOPT of the long-named option found.
1099  * It is only valid when a long-named option has been found by the most
1100  * recent call.
1101  *
1102  * Return the option character from OPTS just read. Return -1 when there are
1103  * no more options. For unrecognized options, or options missing arguments,
1104  * `custom_optopt' is set to the option letter, and '?' is returned.
1105  *
1106  * The OPTS string is a list of characters which are recognized option letters,
1107  * optionally followed by colons, specifying that that letter takes an
1108  * argument, to be placed in `custom_optarg'.
1109  *
1110  * If a letter in OPTS is followed by two colons, its argument is optional.
1111  * This behavior is specific to the GNU `getopt'.
1112  *
1113  * The argument `--' causes premature termination of argument scanning,
1114  * explicitly telling `getopt' that there are no more options. If OPTS begins
1115  * with `--', then non-option arguments are treated as arguments to the option
1116  * '\0'. This behavior is specific to the GNU `getopt'.
1117  */
1118 
1119 static int getopt_internal_r(int argc, char *const *argv, const char *optstring,
1120  const struct option *longopts, int *longind,
1121  struct custom_getopt_data *d)
1122 {
1123  int ret, print_errors = d->custom_opterr;
1124 
1125  if (optstring[0] == ':')
1126  print_errors = 0;
1127  if (argc < 1)
1128  return -1;
1129  d->custom_optarg = NULL;
1130 
1131  /*
1132  * This is a big difference with GNU getopt, since optind == 0
1133  * means initialization while here 1 means first call.
1134  */
1135  if (d->custom_optind == 0 || !d->initialized) {
1136  if (d->custom_optind == 0)
1137  d->custom_optind = 1; /* Don't scan ARGV[0], the program name. */
1138  custom_getopt_initialize(d);
1139  }
1140  if (d->nextchar == NULL || *d->nextchar == '\0') {
1141  ret = shuffle_argv(argc, argv, longopts, d);
1142  if (ret)
1143  return ret;
1144  }
1145  if (longopts && (argv[d->custom_optind][1] == '-' ))
1146  return check_long_opt(argc, argv, optstring, longopts,
1147  longind, print_errors, d);
1148  return check_short_opt(argc, argv, optstring, print_errors, d);
1149 }
